基本GET请求

基本写法

>>> response=requests.get('http://httpbin.org/get')
>>> print(response.text)
{
  "args": {}, 
  "headers": {
    "Accept": "*/*", 
    "Accept-Encoding": "gzip, deflate", 
    "Connection": "close", 
    "Host": "httpbin.org", 
    "User-Agent": "python-requests/2.12.4"
  }, 
  "origin": "223.95.78.229", 
  "url": "http://httpbin.org/get"
}

>>>

带参数GET请求

  • 直接将参数放在URL中进行传递
>>> response=requests.get('http://httpbin.org/get?name=fighter&age=18')
>>> print(response.text)
{
  "args": {
    "age": "18", 
    "name": "fighter"
  }, 
  "headers": {
    "Accept": "*/*", 
    "Accept-Encoding": "gzip, deflate", 
    "Connection": "close", 
    "Host": "httpbin.org", 
    "User-Agent": "python-requests/2.12.4"
  }, 
  "origin": "223.95.78.229", 
  "url": "http://httpbin.org/get?name=fighter&age=18"
}

>>>
  • 也可将数据放在字典中进行传递
>>> response=requests.get('http://httpbin.org/get',params=data)
>>> print(response)
<Response [200]>
>>> print(response.text)
{
  "args": {
    "age": "18",

解析json

  • response.text得到的是str类型的数据结构，用json解析之后得到的是dict类型
>>> import json
>>> a=response.text
>>> b=json.loads(a)
>>> print(b)
{'args': {'age': '18', 'name': 'fighteryu'}, 'url': 'http://httpbin.org/get?age=18&name=fighteryu', 'headers': {'Accept-Encoding': 'gzip, deflate', 'Host': 'httpbin.org', 'User-Agent': 'python-requests/2.12.4', 'Connection': 'close', 'Accept': '*/*'}, 'origin': '223.95.78.229'}
>>> type(a)
<class 'str'>
>>> type(b)
<class 'dict'>
>>>

获取二进制数据

>>> response=requests.get('http://www.fighteryu.xyz/images/icon.ico')

<class 'str'>

添加headers

  • 在没有添加请求头的情况下，返回500
>>> response=requests.get('http://www.zhihu.com/explore')
>>> print(response.text)
<html><body><h1>500 Server Error</h1>
An internal server error occured.
</body></html>

>>>
  • 添加请求头headers后，返回200
>>> headers={"User-Agent":"Mozilla/5.0 (Windows NT 6.1) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/50.0.2661.102 Safari/537.36"}
>>> response=requests.get('http://www.zhihu.com/explore',headers=headers)
>>> response.status_code
200

超时设置

import requests
from requests.exceptions import ReadTimeout
try:
    response = requests.get("http://httpbin.org/get", timeout = 0.5)
    print(response.status_code)
except ReadTimeout:
    print('Timeout')

异常处理

import requests
from requests.exceptions import ReadTimeout, ConnectionError, RequestException
try:
    response = requests.get("http://httpbin.org/get", timeout = 0.5)
    print(response.status_code)
except ReadTimeout:
    print('Timeout')
except ConnectionError:
    print('Connection error')
except RequestException:
    print('Error')
